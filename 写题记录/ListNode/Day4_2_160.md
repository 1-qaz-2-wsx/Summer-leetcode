/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {
    public ListNode getIntersectionNode(ListNode headA, ListNode headB) {
        
        int lenA = 0, lenB = 0;
        ListNode curA = headA;
        ListNode curB = headB;
        // count headA
        while(curA != null){
            lenA++;
            curA = curA.next;
        }

        //count headB
        while(curB != null){
            lenB++;
            curB = curB.next;
        }

        // align
        if(lenB > lenA){
            int temp = lenB;
            lenB = lenA;
            lenA = temp;

            ListNode tmp = headB;
            headB = headA;
            headA = tmp;
        }
        int gap = lenA - lenB;
        curA = headA;
        curB = headB;

        for(int i = 0; i < gap; i++){
            curA = curA.next;
        }

        while(curA != null){
            if(curA == curB){
                return curA;
            }
            curA = curA.next;
            curB = curB.next;
        }

        return null;

    }
}

/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {
    public ListNode getIntersectionNode(ListNode headA, ListNode headB) {
        ListNode p1 = headA;
        ListNode p2 = headB;

        while(p1 != p2){
            if(p1 == null){
                p1 = headB;
            }
            else p1 = p1.next;

            if(p2 == null){
                p2 = headA;
            }
            else p2 = p2.next;
        } 
        return p1;
    }
}
